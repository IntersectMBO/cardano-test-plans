export const CustomTable = ({ columnWidths, tableData }) => {
  // Render function
  return (
    <table style={{ width: '100%' }}>
      <thead>
        <tr>
          {tableData[0].map((header, index) => (
            <th key={index} style={{ width: columnWidths[index] }}>
              {header}
            </th>
          ))}
        </tr>
      </thead>
      <tbody>
        {tableData.slice(1).map((row, rowIndex) => (
          <tr key={rowIndex}>
            {row.map((cell, cellIndex) => (
              <td key={cellIndex} style={{ width: columnWidths[cellIndex] }}>
                {cell.split('\n').map((text, i) => (
                  <span key={i}>
                    {text}
                    {i !== cell.split('\n').length - 1 && <br />}
                  </span>
                ))}
              </td>
            ))}
          </tr>
        ))}
      </tbody>
    </table>
  );
};
